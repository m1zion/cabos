//PARA PODER COMPILARO LA PAGINA Y SE GENEREN PARA EFECTOS DE SEO
1. Se instalo la libreria de react-snap
2. Agregar a package.json las siguientes lineas

IMPORTANTE: Cuando  hacemos el deploy en netlify tenemos que dejar solo  "postbuild": "react-snap", ya que requiere bastantes recursos
"scripts": {
  "build": "vite build",
  "postbuild": "node generate-routes.cjs && react-snap"
},
"reactSnap": {
    "source": "build",
    "destination": "build",
    "crawl": false,
    "skipThirdPartyRequests": true,
    "inlineCss": false,
    "concurrency": 1,
    "routes": []
}
3. Generar el archivo generate-routes.cjs, en este archivo mapeamos las rutas manualmente desde los archivos JSON
  IMPORTANTE 
4. Se agregó un script a index.html
 <script>
  (function () {
    var isSnap = navigator.userAgent === "ReactSnap";
    if (!isSnap && window.location.pathname === "/") {
      var lang = navigator.language.startsWith("es") ? "es" : "en";
      window.location.replace("/" + lang);
    }
  })();
</script>
5. Tambien se configuró el Helmet-provider para agregar metas a las paginas, se agreo en el main.jsx






const fs = require('fs')

const toursData = JSON.parse(
  fs.readFileSync('./src/data/tours.json', 'utf-8')
)
console.log("✅ react-snap config loaded")
const languages = ['en', 'es', 'de']
const routes = ['/']

languages.forEach(lang => {
  routes.push(`/${lang}`)
  toursData.tours.forEach(tour => {
    routes.push(`/${lang}/tours/${tour.category}`)
  })
})

console.log("ROUTES LOADED:", routes)

module.exports = {
  source: 'build',
  destination: 'build',
  routes,
  crawl: false,               // disable crawling — use explicit routes only
  skipThirdPartyRequests: true,
  include: routes,
  inlineCss: false,
  concurrency: 1,
   minifyCss: false
}
